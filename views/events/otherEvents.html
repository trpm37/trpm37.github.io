        //页面滚动
        $(window).on("scroll", function(e){
            var scrollTop = $(window).scrollTop();
            var windowHeight = $(window).height();
            var documentHeight = $(document).height();
            console.log(scrollTop+"|"+windowHeight+"|"+documentHeight+"|"+(scrollTop+windowHeight));
            if (scrollTop + windowHeight == documentHeight) {
                //加载下一页
            }
        });

        //div滚动
        <div id="div-box" style="height:200px;overflow-y:auto;">
            <div class="list-box" style="height:500px;">内容</div>
        </div>
        $("#div-box").scroll(function(){
            var scrollTop=$("#div-box")[0].scrollTop;
            var boxHeight=$("#div-box").height();
            var listBoxHeight=$("#div-box .list-box").height();
            console.log(scrollTop+"|"+boxHeight+"|"+listBoxHeight+"|"+(scrollTop+boxHeight));
            if (scrollTop + boxHeight == listBoxHeight) {
                //加载下一页
            }
        })

        //阻止页面滚动
        document.body.addEventListener('touchmove', function (e) {
          e.preventDefault(); //阻止默认的处理方式(阻止下拉滑动的效果)
        }, {passive: false}); //passive 参数不能省略，用来兼容ios和android

        //关闭当前窗口
        window.onbeforeunload=function(e){
            var e = window.event||e;
            e.returnValue=("确定离开当前页面吗？");
