new Picker(document.querySelector('.js-super-picker'), {
			container: '.js-super-picker-container',
			format: 'YYYY年M月D日H时m分s秒.SSS',
			rows: 4,
			headers: true,
			text: {
				title: '选择日期时间',
				cancel: '取消',
				confirm: '确认',
			},
			translate(type, text) {
				const suffixes = {
					year: '年',
					month: '月',
					day: '日',
					hour: '时',
					minute: '分',
					second: '秒',
					millisecond:'毫秒',
				};
				return Number(text) + suffixes[type];
			},
			increment: {
				minute: 10,
			},
		});
