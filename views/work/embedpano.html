       //加载场景
        function load_pano(obj){
            $("#panoBox").html('');
            var config_myserver = '{$Think.config.think720_config.myserver}';
            var config_mypath = '__TMPL__/_static/home/krpano/';
            var xml_url="{:U('think720/home/doGetPreviewXml')}&id="+obj.media_id+"&vkey="+obj.vkey+"&hmat=0&vmat=0&fmat=100&littleplanetintro=true";
            embedpano({
                html5: "auto",
                target: "panoBox",
                xml: xml_url,
                wmode: "opaque",
                passQueryParameters: true,
                mwheel:true,
                webglsettings:{preserveDrawingBuffer:true},
                initvars:{
                    mypath:config_mypath,
                    time:"20161009",
                    myserver : config_myserver,
                    hmat : -24,
                    vmat : 6,
                    fmat : 70,
                    gyro : true
                }
            });
        }
